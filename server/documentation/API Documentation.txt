Users
	POST "/createUser" = Create a user in the system
		Body: JSON string
			username: string
			password: string

	GET "/logout"

	POST "/subscribe" = Subscribes a device to the notification service
	    Body: JSON string
	        id: string
	            - Registration ID given by FCM

	
Get Stored Information
	GET "/getinfo" = Gets inventory information about a scanned barcode
		URL Parameters: 
			serial: int
	
	GET "/getLastScans" = Gets the most recent scans, up to 10

	GET "/search" = Get a list of items that match a search term
	    Body: JSON string
	        type: string
	            - "name" or "barcode"
	        term: string
	            - Search term
	    Return: JSON array
	        barcodeID: string
	            - Barcode of item
	        title: string
	            - Title of the item
	        price: float
	            - Price of the item
	        onHand: int
	            - Number of item in inventory


Alter Stored Information
	POST "/createItem" = Adds an item to the database
		Body: JSON string
			barcodeID: string
				- Scanned ID of the barcode
			name: string
				- Name of the product
			price: float
				- Price of the product
			minStock: int
				- Minimum number of items left in stock before notification sent to request more inventory
			count: int
				- Amount of product to add
		
	PUT "/editItem" = Edits the details about an item that is stored in the database
		Body: JSON string
			barcodeID: string
				- Scanned ID of the barcode
			name: string
				- Name of the product
			price: float
				- Price of the product
			minStock: int
				- Minimum number of items left in stock before notification sent to request more inventory
			count: int
				- Amount of product to add

	PUT "/editStock" = Edits the inventory count of an item
		Body: JSON string
			count: int
				- Amount of product to add
			barcodeID: string
				- Scanned ID of the barcode
